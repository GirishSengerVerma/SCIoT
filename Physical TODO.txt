Sensors:
	Announce on startup of script:
		TOPIC sensors/metadata/<location>/<measure>
		MESSAGE {
			instanceId,	// "SKP_LIGHT_S", "SMES_WATER_LEVEL_S"
			name,		// "Light at Stuttgart Max Eyth See", "Water Level at Stuttgart Max Eyth See"
			location,	// "STUTTGART_MAX_EYTH_SEE", STUTTGART_KILLESBERG_PARK"
			measure,	// "LIGHT", "WATER_LEVEL"
		}

	Periodically send:
		TOPIC sensors/telemetry/<location>/<measure>
		MESSAGE {
			instanceId,	// "SKP_LIGHT_S", "SMES_WATER_LEVEL_S"
			timestamp, 	// Current ISO Datetime String!! e.g. "2022-07-16T09:59:42.823015+00:00"
			unit,		// "BOOLEAN", "METERS"
			value		// true/false, decimal
		}

Actuators:
	Announce on startup of script:
		TOPIC actuators/metadata/<location>/<type>
		MESSAGE {
			instanceId,	// "SKP_LIGHT_A", "SKP_SOUND_A", "SMES_LIGHT_A", "SMES_WATER_PROTECTION_WALL_A"
			name,		// "Alarm Light at Stuttgart Max Eyth See", "Alarm Sound at Stuttgart Max Eyth See", "Alarm Light at Stuttgart Max Eyth See", "Water Protection Wall at Stuttgart Max Eyth See"
			location,	// "STUTTGART_MAX_EYTH_SEE", "STUTTGART_KILLESBERG_PARK"
			type,		// "ALARM_LIGHT", "ALARM_SOUND", "WATER_PROTECTION_WALL"
		}

	Listen to status messages and update correspondingly:
		TOPIC actuators/status/<location>/<type> 
		MESSAGE {
			instanceId, // "SKP_LIGHT_A", "SKP_SOUND_A", "SMES_LIGHT_A", "SMES_WATER_PROTECTION_WALL_A"
			timestamp,	// Current ISO Datetime String!! e.g. "2022-07-16T09:59:42.823015+00:00"
			enabled,	// true/false
		}
